Use these SQL commands to create databases locally or get copy from Shellb

Teams:

sqlite3 teams.db;
CREATE TABLE teams(
    team_id INT PRIMARY KEY,
    team_name VARCHAR(100),
    conference_id INT,
    conference_name VARCHAR(100),
    coach VARCHAR(100)
);

sqlite3 seasons.db;

CREATE TABLE Seasons (
    team_id INT,
    team_name VARCHAR(250),
    coach_name VARCHAR(250),
    games_won INT,
    games_lost INT,
    season INT,
    PRIMARY KEY (team_id, season)
);

sqlite3 games.db;

CREATE TABLE Games (
    game_id INT,
    season INT,
    neutral_site BOOLEAN,
    home_team_id INT,
    away_team_id INT,
    home_team_name VARCHAR(255),
    away_team_name VARCHAR(255),
    winning_team_id INT,
    winning_team_name VARCHAR(255),
    winning_team_score INT,
    losing_team_id INT,
    losing_team_name VARCHAR(255),
    losing_team_score INT,
    PRIMARY KEY (game_id, season)
);

CREATE TABLE Coaches (
    coach_name VARCHAR(255) PRIMARY KEY,
    start_season INT,
    end_season INT,
    active BOOLEAN,
    current_team VARCHAR(255)
);

CREATE TABLE Schedule (
    game_id INT,
    season INT,
    home_team_id INT,
    away_team_id INT,
    home_team_name VARCHAR(255),
    away_team_name VARCHAR(255),
    PRIMARY KEY(game_id, season)
);

// Simple table to keep track of which teams are in the NIT

CREATE TABLE nit (
    game_id INT,
    season INT,
    PRIMARY KEY(game_id, season)
);

CREATE TABLE inactive_teams (
    team_id INT,
    team_name VARCHAR(100),
    conference_id INT,
    conference_name VARCHAR(100),
    last_active_year INT,
    PRIMARY KEY(team_id, team_name)
);
